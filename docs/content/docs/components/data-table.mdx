## Layout

Import the components and compose them together:

```tsx
import { DataTable } from "@/components/data-table/data-table";
import { DataTableToolbar } from "@/components/data-table/data-table-toolbar";
import { DataTableAdvancedToolbar } from "@/components/data-table/data-table-advanced-toolbar";
import { DataTableFilterList } from "@/components/data-table/data-table-filter-list";
import { DataTableSortList } from "@/components/data-table/data-table-sort-list";
import { useDataTable } from "@/hooks/use-data-table";

const { table } = useDataTable({
  data,
  columns,
  pageCount,
});

// Basic usage
<DataTable
  table={table}
  columns={columns}
  data={data}
/>

// With advanced toolbar
<DataTable table={table}>
  <DataTableAdvancedToolbar table={table}>
    <DataTableFilterList table={table} />
    <DataTableSortList table={table} />
  </DataTableAdvancedToolbar>
</DataTable>

// With standard toolbar
<DataTable table={table}>
  <DataTableToolbar table={table}>
    <DataTableSortList table={table} />
  </DataTableToolbar>
</DataTable>
```

## Column Definitions

```tsx
const columns = React.useMemo(() => [
  {
    // Required: Unique identifier for the column
    id: "title", // [!code highlight]
    // Required: Key to access the data, `accessorFn` can also be used
    accessorKey: "title", // [!code highlight]
    // Optional: Custom header component
    header: ({ column }) => (
      <DataTableColumnHeader column={column} title="Title" />
    ),
    // Optional: Custom cell component
    cell: ({ row }) => <div>{row.getValue("title")}</div>,
    // Optional: Meta options for filtering, sorting, and view options
    meta: {
      label: "Title",
      placeholder: "Search titles...",
      variant: "text",
      icon: Text,
    },
    // Enable features for this column
    enableColumnFilter: true, // [!code highlight]
    enableSorting: true,
    enableHiding: true,
  },
  {
    id: "price",
    // Access nested data using `accessorFn`
    accessorFn: (row) => row.line.price, // [!code highlight]
    header: "Price",
    meta: {
      label: "Price",
      variant: "number",
      icon: DollarSign,
      unit: "$",
    },
    enableColumnFilter: true,
  },
], []);
```

## Walkthrough

<Steps>
  <Step>
    Define columns with appropriate metadata:

    ```tsx
    const columns = React.useMemo(() => [
      {
        // Provide an unique id for the column.
        // This id will be used as query key for the column filter.
        id: "title", // [!code highlight]
        accessorKey: "title",
        header: ({ column }) => (
          <DataTableColumnHeader column={column} title="Title" />
        ),
        cell: ({ row }) => <div>{row.getValue("title")}</div>,
        // Define the column meta options for the filter.
        meta: { // [!code highlight]
          label: "Title", // [!code highlight]
          placeholder: "Search titles...", // [!code highlight]
          variant: "text", // [!code highlight]
          icon: Text, // [!code highlight]
        }, // [!code highlight] 
        // Enable filtering for the column by setting the enableColumnFilter to true.
        enableColumnFilter: true, // [!code highlight]
      },
      {
        id: "status",
        accessorKey: "status",
        header: ({ column }) => (
          <DataTableColumnHeader column={column} title="Status" />
        ),
        cell: ({ cell }) => {
          const status = cell.getValue();
          return (
            <Badge variant="outline">
              <span className="capitalize">{status}</span>
            </Badge>
          );
        },
        // Use variant `select` or `multi-select` with `options` for faceted filtering
        meta: { // [!code highlight]
          label: "Status", // [!code highlight]
          variant: "multiSelect", // [!code highlight]
          options: [ // [!code highlight]
            { label: "Active", value: "active", count: 10 }, // [!code highlight]
            { label: "Inactive", value: "inactive", count: 5 }, // [!code highlight]
          ], // [!code highlight]
        },
        enableColumnFilter: true,
      },
      // Date range column
      {
        id: "createdAt",
        accessorKey: "createdAt",
        header: ({ column }) => (
          <DataTableColumnHeader column={column} title="Created At" />
        ),
        cell: ({ cell }) => formatDate(cell.getValue()),
        // Use variant `date-range` for date range filtering
        meta: { // [!code highlight]
          label: "Created At", // [!code highlight]
          variant: "dateRange", // [!code highlight]
          icon: CalendarIcon, // [!code highlight]
        }, // [!code highlight]
        enableColumnFilter: true,
      },
    ], []);
    ```
  </Step>
  <Step>
    Use the provided hook to initialize your table:

    
  ```tsx
  import { useDataTable } from "@/hooks/use-data-table";

  const { table } = useDataTable({
    data,
    columns,
    // Set the total number of pages for the table.
    pageCount, // [!code highlight]
    initialState: {
      sorting: [{ id: "createdAt", desc: true }],
      columnPinning: { right: ["actions"] },
    },
    // Set the row id for the table for unique row selection.
    getRowId: (row) => row.id, // [!code highlight]
  });
  ```
  </Step>
</Steps>


## Sort List

The [`DataTableSortList`](#datatablesortlist) allows you to sort data by multiple columns simultaneously. Each sort can be configured with its own direction (asc/desc) and can be reordered using drag and drop.

### Installation

```package-install
npx shadcn@latest add "https://diceui.com/r/data-table-sort-list"
```

## Filter List

The [`DataTableFilterList`](#datatablefilterlist) provides a comprehensive way to filter data with multiple conditions. It supports various filter types and operators, with the ability to combine filters using AND/OR logic.

### Installation

```package-install
npx shadcn@latest add "https://diceui.com/r/data-table-filter-list"
```

## Filter Menu

The [`DataTableFilterMenu`](#datatablefiltermenu) provides a command palette-style interface for quickly adding and managing filters. It supports keyboard shortcuts and offers a streamlined way to filter data.

### Installation

```package-install
npx shadcn@latest add "https://diceui.com/r/data-table-filter-menu"
```

## Action Bar

The [`DataTableActionBar`](#datatableactionbar) provides a toolbar for the data table when rows are selected.

### Installation

```package-install
npx shadcn@latest add "https://diceui.com/r/data-table-action-bar"
```
