## Installation

### CLI

```package-install
npx shadcn@latest add "https://diceui.com/r/stat"
```

### Manual

<Steps>
  <Step>
     Install the following dependencies:

     ```package-install
     @radix-ui/react-slot class-variance-authority
     ```
  </Step>
  <Step>
    Add the Separator component if you don't already have it:

    ```package-install
    npx shadcn@latest add "https://diceui.com/r/separator"
    ```
  </Step>
  <Step>
    Copy and paste the following code into your project.

    <ComponentSource name="stat" />
  </Step>

## Layout

```tsx
import {
  Stat,
  StatLabel,
  StatValue,
  StatIndicator,
  StatTrend,
  StatDescription,
  StatSeparator,
} from "@/components/ui/stat";

<Stat>
  <StatLabel>Total Revenue</StatLabel>
  <StatValue>$45,231</StatValue>
  <StatIndicator variant="icon" color="success">
    <DollarSign />
  </StatIndicator>
  <StatTrend trend="up">
    <ArrowUp />
    +20.1% from last month
  </StatTrend>
</Stat>
```
