## Layout

```tsx
import * as ActionBar from "@/components/ui/action-bar";

<ActionBar.Root open={open} onOpenChange={setOpen}>
  <ActionBar.Selection>
    <ActionBar.Close />
  </ActionBar.Selection>
  <ActionBar.Separator />
  <ActionBar.Item />
</ActionBar.Root>
```
