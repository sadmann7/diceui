## Layout

Import the parts, and compose them together.

```tsx
import * as CheckboxGroup from "@diceui/checkbox-group";

<CheckboxGroup.Root>
  <CheckboxGroup.Label />
  <CheckboxGroup.List>
    <CheckboxGroup.Item>
      <CheckboxGroup.Indicator />
    </CheckboxGroup.Item>
  </CheckboxGroup.List>
  <CheckboxGroup.Description>
</CheckboxGroup.Root>
```

## Animated Checkbox

<Steps>
  <Step>
    Update `tailwind.config.ts` to include the following animation:

    ```ts
    export default {
      theme: {
        extend: {
          keyframes: {
            "stroke-dashoffset": {
              "0%": { strokeDashoffset: "100%" },
              "100%": { strokeDashoffset: "0" },
            },
          },
          animation: {
            "stroke-dashoffset": "stroke-dashoffset 0.2s linear forwards",
          },
        },
      },
    }
    ```
  </Step>
  <Step>
    Copy and paste the `CheckboxGroup.Indicator` block from the following example into your project.

    <ComponentTabs name="checkbox-group-animated-demo" />
  </Step>
</Steps>
