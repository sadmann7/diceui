## Installation

### CLI

```package-install
npx shadcn@latest add "https://diceui.com/r/visually-hidden"
```

## Usage

```tsx
import { VisuallyHidden } from "@diceui/visually-hidden"

export default function IconButton() {
  return (
    <button>
      <Icon />
      <VisuallyHidden>Close menu</VisuallyHidden>
    </button>
  )
}
```
