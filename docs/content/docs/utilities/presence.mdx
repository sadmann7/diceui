## Usage

```tsx
import { Presence } from "@/components/presence"

export default function App() {
  const isAuthenticated = false

  return (
    <Presence present={isAuthenticated}>
      <button>Hello, User!</button>
    </Presence>
  )
}
```

When the parameter inserted into "when" prop. will be used inside the children function, in order to have type-safety, pass it on the render function, such as:

```tsx
import { Presence } from "@/components/presence"

type User = {
  name: string
} | null

export default function App() {
  const [user, setUser] = React.useState<User>(null)

  return (
    <Presence present={user}>
      <span>User Name: {user.name}</span>
    </Presence>
  )
}
```

It is also possible to nest Show components:

```tsx
import { Presence } from "@/components/presence"

type User = {
  name: string
  isVIP: boolean
} | null

export default function App() {
  const [user] = useState<User>(null)

  return (
    <Presence present={user}>
      <span>User name: {user.name}</span>
        <span>
          Member Category:
          <Show when={user.isVIP} fallback={<>Member</>}>
            {() => <>VIP</>}
        </Show>
      </span>
    </Presence>
  )
}
```
